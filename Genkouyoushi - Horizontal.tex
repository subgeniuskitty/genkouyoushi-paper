\newcommand\genkoyoshi[4]{% maxrow, maxcol, size, sep
  \def\maxrow{#1}%
  \def\maxcol{#2}%
  \def\size{#3}%
  \def\sep{#4}%
  \begin{tikzpicture}
    \pgfmathsetmacro{\inc}{\size+\sep}
    \foreach \col in {1,...,\maxcol}{
      \foreach \row in {1,...,\maxrow}{
        \draw ({(\col-1)*\inc pt},{\row*\size})
        rectangle ++ (\size,\size);
      }
      \draw ({(\col-1)*\inc-\sep},{\size})
      rectangle ++ (\sep,20*\size);
    }
  \end{tikzpicture}%
}
